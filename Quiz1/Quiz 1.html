agePropData = pd.read_csv("data_quiz/wdi_ageprop_2020.csv")
list_varnames = ["percent_ages0to14","percent_ages15to64"]

for var in list_varnames:
    title = "Distribution for " + var
    plt.hist(x=agePropData[var])
    plt.title(title)
    plt.xlabel(var)
    plt.ylabel("Frequency")
    plt.show()
